16class EventBus: 

17 """Registro in-memory de handlers por tipo de evento. 

18 

19 Uso: 

20 bus = EventBus() 

21 bus.subscribe("AlertaCriado", meu_handler) 

22 bus.publish({"tipo": "AlertaCriado", "cod_alerta": 42, ...}) 

23 """ 

24 

25 def __init__(self) -> None: 

26 self._handlers: dict[str, list[Handler]] = defaultdict(list) 

27 

28 def subscribe(self, tipo: str, handler: Handler) -> None: 

29 self._handlers[tipo].append(handler) 

30 

31 def publish(self, evento: dict) -> None: 

32 """Chama todos os handlers registrados para o tipo do evento. 

33 

34 Erros em handlers individuais são logados mas não interrompem 

35 a entrega aos demais handlers. 

36 """ 

37 tipo = evento.get("tipo", "") 

38 for handler in self._handlers.get(tipo, []): 

39 try: 

40 handler(evento) 

41 except Exception as exc: # noqa: BLE001 

42 logger.error( 

43 "Handler %s falhou para evento %s: %s", 

44 handler.__name__, 

45 tipo, 

46 exc, 

47 ) 

48 

49 def handler_count(self, tipo: str) -> int: 

50 """Utilitário de teste — retorna quantos handlers estão registrados.""" 

51 return len(self._handlers.get(tipo, []))

67 

68 Retorna o número de eventos processados nesta chamada. 

69 Marca processado_em mesmo que handlers falhem — o log registra 

70 a falha, mas o evento não é reentregue (evita loop infinito 

71 em caso de handler com bug permanente). 

72 """ 

73 agora = datetime.now(UTC).isoformat(timespec="seconds") 

74 processados = 0

76 with conectar() as conexao: 

77 cursor = conexao.execute(

79 SELECT id, tipo, agregado_id, payload, schema_versao 

80 FROM eventos 

81 WHERE processado_em IS NULL 

82 ORDER BY criado_em ASC 

83 LIMIT ? 

84 """, 

85 (self._batch_size,), 

86 ) 

87 rows = cursor.fetchall() 

88 

89 for row in rows:

91 try: 

92 payload = json.loads(payload_json) 

93 except json.JSONDecodeError:

96 evento = { 

97 "id": id_, 

98 "tipo": tipo, 

99 "agregado_id": agregado_id, 

100 "payload": payload, 

101 "schema_versao": schema_versao, 

102 } 

103 self._bus.publish(evento)

106 "UPDATE eventos SET processado_em = ?, tentativas = tentativas + 1 WHERE id = ?", 

107 (agora, id_), 

108 )
