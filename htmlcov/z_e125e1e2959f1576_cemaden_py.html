1"""CemadenSource — DataSource concreta para o painel de alertas do CEMADEN. 

2 

3Migra a lógica de normalização de payload e mapeamento para domínio Alerta que 

4vivia em monitor.py (pré-B.1.b). O transporte HTTP (retry/backoff) e o parsing 

5de JSON são compartilhados via `sources/_http.py`. 

6 

7Design decisions registradas em wiki/decisions/decision-record.md: 

8- opener injetável via construtor (testes explícitos, sem patch por string) 

9- url injetável via construtor (futuro: staging, mock server) 

10- timeout_segundos injetável (futuro: ajuste por ambiente) 

11- keyword-only no __init__ (legibilidade no call site) 

12 

13Invariantes do contrato CEMADEN: 

14- coletar() captura APENAS ValueError ao montar alerta. Bugs internos 

15 (TypeError, AttributeError, KeyError) propagam para diagnóstico. 

16- Falhas de rodada (rede, HTTPError, JSON inválido) sobem como FalhaDeColeta 

17 com fonte=CEMADEN — geradas em sources/_http.py. 

18""" 

19

41class CemadenSource(DataSource): 

42 """DataSource para o painel de alertas do CEMADEN. 

43 

44 Construtor keyword-only por legibilidade no call site: 

45 CemadenSource() # produção 

46 CemadenSource(opener=fake) # teste 

47 CemadenSource(url=staging_url) # ambiente alternativo 

48 """

50 def __init__( 

51 self, 

52 *, 

53 url: str = URL_CEMADEN, 

54 opener: Opener = urlopen, # type: ignore[assignment] 

55 timeout_segundos: float = TIMEOUT_SEGUNDOS, 

56 ) -> None: 

57 self._url = url 

58 self._opener = opener 

59 self._timeout_segundos = timeout_segundos 

60 

61 @property 

62 def fonte(self) -> FonteDado: 

63 return FonteDado.CEMADEN 

64 

65 def coletar(self) -> ResultadoColeta: 

66 raw = fetch_com_retry( 

67 self._url, 

68 fonte=self.fonte, 

69 opener=self._opener, 

70 user_agent=USER_AGENT, 

71 timeout_segundos=self._timeout_segundos, 

72 ) 

73 payload = parse_json(raw, fonte=self.fonte) 

74 

75 itens_brutos = self._normalize_payload(payload) 

76 alertas: list[Alerta] = [] 

77 descartados = 0 

78 for item in itens_brutos: 

79 try: 

80 alertas.append(self._montar_alerta(item)) 

81 except ValueError: 

82 descartados += 1 

83 # Outras exceções (TypeError, AttributeError, KeyError) propagam: são bug. 

84 

85 return ResultadoColeta( 

86 alertas=alertas, 

87 descartados=descartados, 

88 coletado_em=datetime.now(timezone.utc), 

89 ) 

90 

91 def _normalize_payload(self, payload: object) -> list[dict]: 

92 if isinstance(payload, list): 

93 return payload 

94 if isinstance(payload, dict): 

95 for key in ("alertas", "data", "items", "results"): 

96 value = payload.get(key) 

97 if isinstance(value, list): 

98 return value 

99 raise FalhaDeColeta( 

100 fonte=self.fonte, 

101 causa="formato de payload não reconhecido: dict sem chave " 

102 "conhecida ('alertas', 'data', 'items', 'results') contendo uma lista", 

103 ) 

104 raise FalhaDeColeta( 

105 fonte=self.fonte, 

106 causa=f"formato de payload inesperado: esperado list ou dict, " 

107 f"recebido {type(payload).__name__}", 

108 ) 

109 

110 def _montar_alerta(self, item: dict) -> Alerta: 

111 """Converte item bruto do CEMADEN em Alerta de domínio. 

112 

113 Levanta ValueError se item não tiver campos obrigatórios. Captura em 

114 coletar() conta como descartado; bugs internos (TypeError etc.) propagam. 

115 """ 

116 if not isinstance(item, dict): 

117 raise ValueError(f"item deve ser dict, recebido {type(item).__name__}") 

118 alerta = Alerta.from_dict(item, fonte=self.fonte) 

119 escopo = classificar_escopo(alerta.coordenadas) 

120 

121 categoria = self._categoria_do_evento(item) 

122 if categoria: 

123 tipo_evento = TipoEvento.from_string(categoria) 

124 cobrade = mapear_cemaden(categoria) 

125 else: 

126 # Sem a chave real `evento`: preserva o que from_dict já resolveu 

127 # via seus próprios aliases (fallback defensivo). Sem `evento`, 

128 # não há categoria granular o bastante pra mapear COBRADE. 

129 tipo_evento = alerta.tipo_evento 

130 cobrade = None 

131 if cobrade is not None: 

132 fonte_classificacao = FonteClassificacao.MAPEADA_POR_NOME 

133 else: 

134 fonte_classificacao = FonteClassificacao.INDETERMINADA 

135 

136 return alerta.model_copy( 

137 update={ 

138 "tipo_evento": tipo_evento, 

139 "escopo_geografico": escopo, 

140 "cobrade_codigo": cobrade, 

141 "fonte_classificacao": fonte_classificacao, 

142 } 

143 ) 

144 

145 @staticmethod 

146 def _categoria_do_evento(item: dict) -> str: 

147 """Extrai a categoria bruta do campo `evento` do CEMADEN. 

148 

149 O payload real entrega `evento` como string composta 

150 "<categoria> - <nível>" (ex.: "Risco Hidrológico - Moderado", 

151 confirmado em 475 itens reais de data/samples/cemaden_raw_*.json, 

152 issue #30). Corta no primeiro " - " e retorna só a categoria — é 

153 isso que `TipoEvento.from_string` e `mapear_cemaden` esperam. 

154 Retorna "" se `evento` estiver ausente ou vazio (cai em 

155 TipoEvento.INDETERMINADO / cobrade None, sem inventar). 

156 """ 

157 evento_bruto = item.get("evento") 

158 if not isinstance(evento_bruto, str) or not evento_bruto.strip(): 

159 return "" 

160 return evento_bruto.split(" - ", 1)[0].strip()
