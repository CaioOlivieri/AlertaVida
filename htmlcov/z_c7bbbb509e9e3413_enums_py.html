15def _normalizar(s: str) -> str: 

16 normalized = unicodedata.normalize("NFKD", s.strip().lower()) 

17 return "".join(ch for ch in normalized if not unicodedata.combining(ch))

20class NivelRisco(str, Enum): 

21 BAIXO = "BAIXO" 

22 MODERADO = "MODERADO" 

23 ALTO = "ALTO" 

24 MUITO_ALTO = "MUITO_ALTO" 

25 INDETERMINADO = "INDETERMINADO" 

26 

27 @classmethod 

28 def from_string(cls, valor: str | None) -> "NivelRisco": 

29 if valor is None or not valor.strip(): 

30 raise ValueError("Nível de risco ausente ou inválido") 

31 

32 normalized = _normalizar(valor).replace(" ", "_") 

33 mapping = { 

34 "baixo": cls.BAIXO, 

35 "moderado": cls.MODERADO, 

36 "alto": cls.ALTO, 

37 "muito_alto": cls.MUITO_ALTO, 

38 "muitoalto": cls.MUITO_ALTO, 

39 "indeterminado": cls.INDETERMINADO, 

40 } 

41 if normalized in mapping: 

42 return mapping[normalized] 

43 

44 raise ValueError(f"Nível de risco desconhecido: {valor}")

145class FonteDado(StrEnum): 

146 """Fontes de dados de alertas suportadas pelo sistema. 

147 

148 Valores fechados — typo em string solta violaria UNIQUE(fonte, cod_alerta) 

149 silenciosamente. Cada DataSource (Camada 4 Parte B+) declara sua fonte 

150 via atributo `fonte: FonteDado` no modelo `Alerta`. 

151 

152 Diferente de outros enums do domínio, NÃO inclui INDETERMINADA — fonte 

153 sempre é conhecida no momento de coleta (a `DataSource` que produziu o 

154 `Alerta` declara sua origem). Alerta sem fonte é cenário inválido por 

155 construção. 

156 """ 

157 

158 CEMADEN = "CEMADEN" 

159 EONET = "EONET" 

160 INMET = "INMET" 

161 INPE = "INPE" 

162 

163 @classmethod 

164 def from_string(cls, valor: str | None) -> "FonteDado": 

165 """Normaliza string para FonteDado. Strict — levanta em valor inválido. 

166 

167 Comportamento alinhado com NivelRisco.from_string (levanta em valor 

168 desconhecido), não com TipoEvento.from_string (que retorna sentinela 

169 INDETERMINADO). Justificativa: fonte desconhecida em runtime é bug 

170 grave — banco gravaria valor inválido violando UNIQUE constraint. 

171 Levantar força tratar. 

172 

173 Aceita variações de case e whitespace ('cemaden', 'Cemaden ', 

174 'CEMADEN' todos viram FonteDado.CEMADEN). 

175 """ 

176 if valor is None or not str(valor).strip(): 

177 raise ValueError("Fonte ausente ou vazia") 

178 try: 

179 return cls(str(valor).strip().upper()) 

180 except ValueError as err: 

181 raise ValueError( 

182 f"Fonte desconhecida: {valor!r}. " 

183 f"Válidas: {[f.value for f in cls]}" 

184 ) from err
