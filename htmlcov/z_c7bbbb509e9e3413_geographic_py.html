1"""Classificação geográfica de alertas — Camada 4 Parte A.1.3. 

2 

3Determina o escopo (BRASIL, PROXIMO, INTERNACIONAL, INDETERMINADO) de um alerta 

4a partir das suas coordenadas. Usa bbox + buffer configurável em vez de polígono 

5real (shapely) — quatro comparações numéricas, sem dependência nova, imprecisão 

6documentada e substituível. 

7 

8A faixa BRASIL é fixa (território + pequena margem para alertas costeiros e 

9fronteiriços). A faixa PROXIMO recebe um buffer configurável via env var 

10`ALERTAVIDA_BUFFER_PROXIMO_GRAUS` (default 5° ≈ 500 km). 

11 

12Mudanças no buffer só afetam alertas NOVOS. Re-classificação de alertas 

13existentes é responsabilidade de `scripts/reclassificar_escopos.py` 

14(implementado em A.1.4). 

15""" 

16 

17from __future__ import annotations 

18 

19import os 

20from dataclasses import dataclass 

21 

22from alertavida.domain.coordenadas import Coordenadas 

23from alertavida.domain.enums import EscopoGeografico 

24 

25BUFFER_PROXIMO_DEFAULT_GRAUS: float = 5.0 

26ENV_BUFFER_PROXIMO: str = "ALERTAVIDA_BUFFER_PROXIMO_GRAUS" 

27 

28 

29@dataclass(frozen=True) 

30class FaixaGeografica: 

31 """Bounding box geográfico em graus decimais (WGS84). 

32 

33 Convenção: latitudes negativas no Sul, longitudes negativas no Oeste. 

34 Brasil: aproximadamente lat [-34, +5], lon [-74, -34]. 

35 """ 

36 

37 lat_min: float 

38 lat_max: float 

39 lon_min: float 

40 lon_max: float 

41 

42 def contem(self, coord: Coordenadas) -> bool: 

43 return ( 

44 self.lat_min <= coord.latitude <= self.lat_max 

45 and self.lon_min <= coord.longitude <= self.lon_max 

46 ) 

47 

48 

49# Bbox BRASIL inclui pequena margem para zonas marítimas costeiras e 

50# áreas fronteiriças (Pantanal, Amazônia ocidental). Valores fixos por 

51# decisão arquitetural — alterá-los é mudança de contrato, não config. 

52FAIXA_BRASIL: FaixaGeografica = FaixaGeografica( 

53 lat_min=-34.0, 

54 lat_max=6.0, 

55 lon_min=-74.0, 

56 lon_max=-34.0, 

57) 

58 

59 

60def _ler_buffer_proximo() -> float: 

61 """Lê o buffer PROXIMO da env var, com fallback para o default. 

62 

63 Buffer inválido (não-numérico, negativo ou zero) cai para o default, 

64 sem levantar erro — operações de classificação não devem falhar por 

65 config malformada. 

66 """ 

67 valor_raw = os.getenv(ENV_BUFFER_PROXIMO) 

68 if valor_raw is None or not valor_raw.strip(): 

69 return BUFFER_PROXIMO_DEFAULT_GRAUS 

70 try: 

71 valor = float(valor_raw) 

72 except ValueError: 

73 return BUFFER_PROXIMO_DEFAULT_GRAUS 

74 if valor <= 0: 

75 return BUFFER_PROXIMO_DEFAULT_GRAUS 

76 return valor 

77 

78 

79def _faixa_proximo() -> FaixaGeografica: 

80 """Faixa PROXIMO = bbox BRASIL expandido em todas as direções pelo buffer.""" 

81 buffer = _ler_buffer_proximo() 

82 return FaixaGeografica( 

83 lat_min=FAIXA_BRASIL.lat_min - buffer, 

84 lat_max=FAIXA_BRASIL.lat_max + buffer, 

85 lon_min=FAIXA_BRASIL.lon_min - buffer, 

86 lon_max=FAIXA_BRASIL.lon_max + buffer, 

87 ) 

88 

89 

90def classificar_escopo(coordenadas: Coordenadas | None) -> EscopoGeografico: 

91 """Classifica o escopo geográfico de um alerta. 

92 

93 - Sem coordenadas → INDETERMINADO (default seguro, não chuta). 

94 - Dentro do bbox BRASIL → BRASIL. 

95 - Fora do BRASIL mas dentro do bbox PROXIMO (BRASIL + buffer) → PROXIMO. 

96 - Fora do bbox PROXIMO → INTERNACIONAL. 

97 

98 Buffer da faixa PROXIMO é lido a cada chamada — env vars podem mudar em 

99 runtime (testes, hot-reload). Custo é desprezível (uma leitura de env). 

100 """ 

101 if coordenadas is None: 

102 return EscopoGeografico.INDETERMINADO 

103 if FAIXA_BRASIL.contem(coordenadas): 

104 return EscopoGeografico.BRASIL 

105 if _faixa_proximo().contem(coordenadas): 

106 return EscopoGeografico.PROXIMO 

107 return EscopoGeografico.INTERNACIONAL
