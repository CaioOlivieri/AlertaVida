1"""Transporte HTTP compartilhado entre DataSources — Camada 4 (consolidação C.x). 

2 

3Antes desta extração, CemadenSource e NasaEonetSource duplicavam ~50 linhas 

4idênticas de retry/backoff e parsing de JSON. Centralizar aqui garante uma única 

5implementação da política de resiliência (invariantes 3, 19, 20): 

6 

7- retry apenas em 5xx / 408 / 429 / URLError / socket.timeout; 

8- HTTPError 4xx (exceto 408/429) falha imediatamente, sem retry; 

9- toda falha de transporte/parsing vira FalhaDeColeta(fonte=...) com a exceção 

10 original encadeada — nenhuma exceção crua de rede vaza para o orquestrador. 

11"""

47def fetch_com_retry( 

48 url: str, 

49 *, 

50 fonte: FonteDado, 

51 opener: Opener, 

52 user_agent: str, 

53 timeout_segundos: float = TIMEOUT_SEGUNDOS, 

54 max_tentativas: int = MAX_TENTATIVAS, 

55 backoff_inicial: float = BACKOFF_INICIAL, 

56) -> bytes: 

57 """GET com retry/backoff exponencial. Levanta FalhaDeColeta ao esgotar. 

58 

59 Retry apenas em 5xx / 408 / 429 / URLError / socket.timeout. HTTPError 4xx 

60 (exceto 408/429) vira FalhaDeColeta imediatamente, sem retry. Esgotadas as 

61 tentativas, a última exceção vira FalhaDeColeta com `original` encadeada. 

62 """ 

63 request = Request(url, headers={"User-Agent": user_agent}) 

64 ultima_excecao: Exception | None = None 

65 

66 for tentativa in range(max_tentativas): 

67 tentativa_humana = tentativa + 1 

68 logger.info("[Tentativa %s/%s]", tentativa_humana, max_tentativas) 

69 try: 

70 with opener(request, timeout=timeout_segundos) as response: 

71 return response.read() 

72 except HTTPError as exc: 

73 ultima_excecao = exc 

74 if 400 <= exc.code < 500 and exc.code not in (408, 429): 

75 raise FalhaDeColeta( 

76 fonte=fonte, causa=f"HTTPError {exc.code}", original=exc 

77 ) from exc 

78 except (URLError, socket.timeout) as exc: 

79 ultima_excecao = exc 

80 

81 if tentativa_humana < max_tentativas: 

82 espera = backoff_inicial * (2**tentativa) 

83 logger.warning("Aguardando %gs antes da próxima tentativa...", espera) 

84 time.sleep(espera) 

85 

86 assert ultima_excecao is not None 

87 if isinstance(ultima_excecao, HTTPError): 

88 causa = f"HTTPError {ultima_excecao.code}" 

89 else: 

90 causa = f"rede esgotada após {max_tentativas} tentativas" 

91 raise FalhaDeColeta(fonte=fonte, causa=causa, original=ultima_excecao) from ultima_excecao

94def parse_json(raw: bytes, *, fonte: FonteDado) -> object: 

95 """Decodifica UTF-8 + json.loads. Levanta FalhaDeColeta em falha de parsing.""" 

96 try: 

97 return json.loads(raw.decode("utf-8")) 

98 except UnicodeDecodeError as exc: 

99 raise FalhaDeColeta(fonte=fonte, causa="response não é UTF-8", original=exc) from exc 

100 except json.JSONDecodeError as exc: 

101 raise FalhaDeColeta(fonte=fonte, causa="response não é JSON válido", original=exc) from exc
