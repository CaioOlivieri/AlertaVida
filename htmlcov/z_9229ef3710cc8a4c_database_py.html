1"""Persistência SQLite — Camada 1 + refator Camada 4 Parte A.1.4. 

2 

3Schema da tabela `alertas` reescrito para multi-fonte: 

4- Surrogate key `id` (FK opaca em outras tabelas). 

5- `UNIQUE (fonte, cod_alerta)` substitui PK composta. 

6- Coluna `fonte` discrimina origem do alerta (CEMADEN, EONET, INMET, INPE). 

7- Coluna `escopo_geografico` armazena classificação calculada na ingestão. 

8 

9Tabela `eventos` (Outbox Pattern) preserva contrato — `agregado_id` agora 

10referencia `alertas.id` (surrogate INTEGER), não `cod_alerta` da fonte. 

11O `cod_alerta` original fica preservado dentro do `payload` JSON do evento. 

12"""

40class SchemaIncompativelError(Exception): 

41 """Banco existente tem schema incompatível com a versão atual do código. 

42 

43 Levantada quando criar_banco() detecta uma tabela `alertas` pré-A.1 

44 (sem coluna `id` ou `fonte`). Bancos pré-A.1 não têm caminho de migration 

45 automática — a Camada 4 Parte A.1 (09/05/2026) introduziu ruptura de 

46 schema (PK composta -> surrogate key) que SQLite não suporta via ALTER. 

47 """

50def _verificar_compatibilidade_schema(conexao: sqlite3.Connection) -> None: 

51 """Verifica se o schema existente é compatível com a versão atual. 

52 

53 Casos: 

54 - Tabela `alertas` não existe -> OK, criar_banco() vai criá-la 

55 - Tabela existe com `id` + `fonte` -> OK, _migrar_banco() cuida de aditivos 

56 - Tabela existe sem `id` ou sem `fonte` -> SchemaIncompativelError 

57 

58 Pré-condição: chamada como primeira operação dentro de criar_banco(), 

59 antes de qualquer CREATE TABLE ou ALTER TABLE. 

60 """ 

61 cursor = conexao.execute("PRAGMA table_info(alertas)") 

62 colunas = {row[1] for row in cursor.fetchall()} 

63 

64 if not colunas: 

65 return 

66 

67 colunas_obrigatorias = {"id", "fonte"} 

68 faltantes = colunas_obrigatorias - colunas 

69 

70 if faltantes: 

71 raise SchemaIncompativelError( 

72 f"Schema do banco em '{DB_PATH}' é incompatível com a versão atual.\n" 

73 f"\n" 

74 f"Detectado: tabela `alertas` sem coluna(s): {sorted(faltantes)}.\n" 

75 f"Provável origem: banco criado antes da Camada 4 Parte A.1 (09/05/2026).\n" 

76 f"\n" 

77 f"A Camada 4 Parte A.1 introduziu ruptura estrutural (surrogate key + " 

78 f"UNIQUE composto) sem caminho de migration automática — bancos pré-A.1 " 

79 f"precisam ser recriados.\n" 

80 f"\n" 

81 f"Ação: apague o arquivo do banco e rode criar_banco() novamente.\n" 

82 f"Se houver dados a preservar, exporte para JSON antes de apagar." 

83 )

86def _migrar_banco(conexao: sqlite3.Connection) -> None: 

87 """Aplica migrations aditivas idempotentes ao schema existente. 

88 

89 Hoje cobre as colunas COBRADE da Camada 4 Parte A.2 (`cobrade_codigo`, 

90 `fonte_classificacao`), adicionadas via `ALTER TABLE` apenas quando 

91 ausentes. A migration de PK composta para surrogate (A.1.4) nunca passou 

92 por aqui — o banco estava vazio no refator e rupturas estruturais são 

93 barradas antes por `_verificar_compatibilidade_schema`. É o ponto de 

94 extensão obrigatório para qualquer mudança de schema aditiva futura. 

95 """ 

96 # Camada 4 / A.2 — colunas COBRADE 

97 cursor = conexao.execute("PRAGMA table_info(alertas)") 

98 colunas_existentes = {row[1] for row in cursor.fetchall()} 

99 

100 if "cobrade_codigo" not in colunas_existentes: 

101 conexao.execute("ALTER TABLE alertas ADD COLUMN cobrade_codigo TEXT NULL") 

102 

103 if "fonte_classificacao" not in colunas_existentes: 

104 conexao.execute( 

105 "ALTER TABLE alertas ADD COLUMN fonte_classificacao " 

106 "TEXT NOT NULL DEFAULT 'INDETERMINADA'" 

107 )

110def criar_banco() -> None: 

111 with conectar() as conexao: 

112 conexao.execute("PRAGMA journal_mode=WAL") 

113 _verificar_compatibilidade_schema(conexao) # detecta bancos pré-A.1 

114 conexao.execute( 

115 """ 

116 CREATE TABLE IF NOT EXISTS alertas ( 

117 id INTEGER PRIMARY KEY AUTOINCREMENT, 

118 fonte TEXT NOT NULL DEFAULT 'CEMADEN', 

119 cod_alerta TEXT NOT NULL, 

120 municipio TEXT, 

121 uf TEXT, 

122 evento TEXT, 

123 nivel TEXT, 

124 datahoracriacao TEXT, 

125 detectado_em TEXT NOT NULL, 

126 codibge INTEGER, 

127 latitude REAL NOT NULL, 

128 longitude REAL NOT NULL, 

129 escopo_geografico TEXT NOT NULL DEFAULT 'INDETERMINADO', 

130 ult_atualizacao TEXT, 

131 status_interno TEXT NOT NULL DEFAULT 'ATIVO', 

132 visto_ultima_vez TEXT NOT NULL DEFAULT '', 

133 rodadas_ausente INTEGER NOT NULL DEFAULT 0, 

134 assinatura TEXT, 

135 cobrade_codigo TEXT NULL, 

136 fonte_classificacao TEXT NOT NULL DEFAULT 'INDETERMINADA', 

137 UNIQUE (fonte, cod_alerta) 

138 ) 

139 """ 

140 ) 

141 conexao.execute("CREATE INDEX IF NOT EXISTS idx_uf ON alertas (uf)") 

142 conexao.execute("CREATE INDEX IF NOT EXISTS idx_evento ON alertas (evento)") 

143 conexao.execute("CREATE INDEX IF NOT EXISTS idx_nivel ON alertas (nivel)") 

144 conexao.execute("CREATE INDEX IF NOT EXISTS idx_fonte ON alertas (fonte)") 

145 conexao.execute( 

146 "CREATE INDEX IF NOT EXISTS idx_escopo_geografico ON alertas (escopo_geografico)" 

147 ) 

148 conexao.execute( 

149 """ 

150 CREATE TABLE IF NOT EXISTS eventos ( 

151 id INTEGER PRIMARY KEY AUTOINCREMENT, 

152 tipo TEXT NOT NULL, 

153 agregado_id INTEGER NOT NULL, 

154 payload TEXT NOT NULL, 

155 schema_versao INTEGER NOT NULL DEFAULT 1, 

156 criado_em TEXT NOT NULL, 

157 processado_em TEXT NULL, 

158 tentativas INTEGER NOT NULL DEFAULT 0 

159 ) 

160 """ 

161 ) 

162 conexao.execute( 

163 """ 

164 CREATE INDEX IF NOT EXISTS idx_eventos_pendentes 

165 ON eventos (processado_em, criado_em) 

166 """ 

167 ) 

168 _migrar_banco(conexao) 

169 conexao.commit()

172def buscar_snapshots(fonte: FonteDado) -> list[AlertaSnapshot]: 

173 """Retorna snapshots de todos os alertas (qualquer status) da fonte. 

174 

175 Cada snapshot carrega sua `fonte` como campo (lido da row, não 

176 hardcoded do parâmetro) — robustez contra futuras mudanças no 

177 WHERE da query. Removeu filtro `status_interno = 'ATIVO'` para que 

178 alertas RESOLVIDO que reaparecem sejam detectados corretamente. 

179 """ 

180 with conectar() as conexao: 

181 cursor = conexao.execute( 

182 """ 

183 SELECT cod_alerta, fonte, nivel, evento, ult_atualizacao, 

184 rodadas_ausente, status_interno 

185 FROM alertas 

186 WHERE fonte = ? 

187 """, 

188 (fonte.value,), 

189 ) 

190 return [ 

191 AlertaSnapshot( 

192 cod_alerta=row[0], 

193 fonte=FonteDado.from_string(row[1]), 

194 nivel_risco=row[2], 

195 tipo_evento=row[3], 

196 ult_atualizacao=row[4], 

197 rodadas_ausente=row[5], 

198 status_interno=row[6], 

199 ) 

200 for row in cursor.fetchall() 

201 ]

204def _executar_retornando_id(conexao: sqlite3.Connection, sql: str, params: tuple) -> int | None: 

205 """Executa um UPDATE ... RETURNING id e devolve o id (None se 0 linhas). 

206 

207 Substitui o par UPDATE + SELECT id por uma única query. RETURNING exige 

208 SQLite >= 3.35 (Python 3.13 embute >= 3.40). 

209 """ 

210 row = conexao.execute(sql, params).fetchone() 

211 return row[0] if row is not None else None

214def aplicar_resultado_deteccao( 

215 resultado: ResultadoDeteccao, 

216 alertas_por_codigo: dict[str, "Alerta"], 

217 agora: str, 

218) -> None: 

219 """Persiste o resultado do ChangeDetector atomicamente. 

220 

221 INSERT/UPDATE em alertas e INSERT em eventos ocorrem na mesma transação 

222 SQLite — outbox pattern. `agregado_id` em eventos referencia o `id` 

223 surrogate da tabela `alertas`. 

224 

225 A fonte de cada código é obtida via `resultado.fonte_por_codigo[cod]` 

226 (populado pelo detector) — não recebe `fonte` como parâmetro. Isso 

227 permite que rodadas multi-fonte (Camada 5+) sejam tratadas sem 

228 mudanças nesta função. 

229 """ 

230 with conectar() as conexao: 

231 eventos: list[EventoDetectado] = resultado.eventos 

232 for evento in eventos: 

233 agregado_id: int | None = None 

234 

235 if evento.tipo is TipoEventoDetectado.CRIADO: 

236 alerta = alertas_por_codigo[evento.cod_alerta] 

237 ult = alerta.ult_atualizacao.isoformat() if alerta.ult_atualizacao else None 

238 cursor = conexao.execute( 

239 """ 

240 INSERT INTO alertas ( 

241 fonte, cod_alerta, municipio, uf, evento, nivel, 

242 datahoracriacao, detectado_em, codibge, 

243 latitude, longitude, escopo_geografico, ult_atualizacao, 

244 status_interno, visto_ultima_vez, rodadas_ausente, 

245 assinatura, cobrade_codigo, fonte_classificacao 

246 ) VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, 'ATIVO', ?, 0, NULL, ?, ?) 

247 """, 

248 ( 

249 alerta.fonte.value, 

250 alerta.cod_alerta, 

251 alerta.municipio.nome if alerta.municipio is not None else None, 

252 alerta.municipio.uf if alerta.municipio is not None else None, 

253 alerta.tipo_evento.value, 

254 alerta.nivel_risco.value, 

255 alerta.data_criacao.isoformat(), 

256 agora, 

257 alerta.municipio.codigo_ibge if alerta.municipio is not None else None, 

258 alerta.coordenadas.latitude, 

259 alerta.coordenadas.longitude, 

260 alerta.escopo_geografico.value, 

261 ult, 

262 agora, 

263 alerta.cobrade_codigo, 

264 alerta.fonte_classificacao.value, 

265 ), 

266 ) 

267 agregado_id = cursor.lastrowid 

268 

269 elif evento.tipo in ( 

270 TipoEventoDetectado.ATUALIZADO, 

271 TipoEventoDetectado.REATIVADO, 

272 ): 

273 alerta = alertas_por_codigo[evento.cod_alerta] 

274 ult = alerta.ult_atualizacao.isoformat() if alerta.ult_atualizacao else None 

275 # REATIVADO reativa a linha (status volta a ATIVO); ATUALIZADO 

276 # mantém o status. set_status é literal constante (não entrada 

277 # externa) — sem risco de injeção SQL. 

278 set_status = ( 

279 "status_interno = 'ATIVO', " 

280 if evento.tipo is TipoEventoDetectado.REATIVADO 

281 else "" 

282 ) 

283 agregado_id = _executar_retornando_id( 

284 conexao, 

285 f""" 

286 UPDATE alertas 

287 SET {set_status}nivel = ?, evento = ?, ult_atualizacao = ?, 

288 visto_ultima_vez = ?, rodadas_ausente = 0, 

289 cobrade_codigo = ?, fonte_classificacao = ? 

290 WHERE fonte = ? AND cod_alerta = ? 

291 RETURNING id 

292 """, 

293 ( 

294 alerta.nivel_risco.value, 

295 alerta.tipo_evento.value, 

296 ult, 

297 agora, 

298 alerta.cobrade_codigo, 

299 alerta.fonte_classificacao.value, 

300 alerta.fonte.value, 

301 evento.cod_alerta, 

302 ), 

303 ) 

304 

305 elif evento.tipo is TipoEventoDetectado.RESOLVIDO: 

306 agregado_id = _executar_retornando_id( 

307 conexao, 

308 """ 

309 UPDATE alertas 

310 SET status_interno = 'RESOLVIDO', visto_ultima_vez = ? 

311 WHERE fonte = ? AND cod_alerta = ? 

312 RETURNING id 

313 """, 

314 (agora, evento.fonte.value, evento.cod_alerta), 

315 ) 

316 

317 if agregado_id is not None: 

318 conexao.execute( 

319 """ 

320 INSERT INTO eventos ( 

321 tipo, agregado_id, payload, schema_versao, 

322 criado_em, processado_em, tentativas 

323 ) VALUES (?, ?, ?, 1, ?, NULL, 0) 

324 """, 

325 ( 

326 evento.tipo, 

327 agregado_id, 

328 json.dumps(evento.payload, ensure_ascii=False), 

329 agora, 

330 ), 

331 ) 

332 

333 codigos_com_evento = {e.cod_alerta for e in resultado.eventos} 

334 for cod in resultado.codigos_vistos - codigos_com_evento: 

335 fonte_cod = resultado.fonte_por_codigo[cod] 

336 conexao.execute( 

337 """ 

338 UPDATE alertas 

339 SET visto_ultima_vez = ?, rodadas_ausente = 0 

340 WHERE fonte = ? AND cod_alerta = ? 

341 """, 

342 (agora, fonte_cod.value, cod), 

343 ) 

344 

345 for cod in resultado.codigos_ausentes: 

346 fonte_cod = resultado.fonte_por_codigo[cod] 

347 conexao.execute( 

348 """ 

349 UPDATE alertas 

350 SET rodadas_ausente = rodadas_ausente + 1, 

351 visto_ultima_vez = ? 

352 WHERE fonte = ? AND cod_alerta = ? 

353 """, 

354 (agora, fonte_cod.value, cod), 

355 ) 

356 

357 conexao.commit()
